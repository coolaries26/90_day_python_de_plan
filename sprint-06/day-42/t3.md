        c.cluster,
        count(*) as customer_count,
        round(avg(a.total_spend)::numeric, 2) as avg_spend,
        round(avg(a.total_rentals)::numeric, 2) as avg_rentals
    from analytics_customer_clusters c
    join analytics_customer_airflow a on c.customer_id = a.customer_id
    group by c.cluster
    order by avg_spend desc
